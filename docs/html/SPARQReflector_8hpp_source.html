43 static std::vector<std::string> extractBindings(std::string_view json) {
44 std::vector<std::string> bindings;
45
46 // 1. Find "results"
47 size_t resultsPos = json.find("\"results\"");
48 if (resultsPos == std::string_view::npos) return bindings;
49
50 // 2. Find "bindings" inside results
51 size_t bindingsPos = json.find("\"bindings\"", resultsPos);
52 if (bindingsPos == std::string_view::npos) return bindings;
53
54 // 3. Find the start of the array '['
55 size_t arrayStart = json.find('[', bindingsPos);
56 if (arrayStart == std::string_view::npos) return bindings;
57
58 // 4. Iterate through array elements (objects enclosed in { })
59 size_t currentPos = arrayStart + 1;
60 int braceCount = 0;
61 size_t itemStart = std::string_view::npos;
62
63 while (currentPos < json.length()) {
64 char c = json[currentPos];
65 if (c == '{') {
66 if (braceCount == 0) itemStart = currentPos;
67 braceCount++;
68 } else if (c == '}') {
69 braceCount--;
70 if (braceCount == 0 && itemStart != std::string_view::npos) {
71 // Found a complete object { ... }
72 bindings.emplace_back(json.substr(itemStart, currentPos - itemStart + 1));
73 itemStart = std::string_view::npos;
74 }
75 } else if (c == ']' && braceCount == 0) {
76 break; // End of bindings array
77 }
78 currentPos++;
79 }
80 return bindings;
81 }

91 static std::string extractValue(std::string_view rowJson, std::string_view varName) {
92 // Construct search key like "varName"
93 std::string key = "\"" + std::string(varName) + "\"";
94
95 size_t keyPos = rowJson.find(key);
96 if (keyPos == std::string_view::npos) return "";
97
98 // Find the opening brace of this variable's object '{'
99 // Skip whitespace after key
100 size_t objStart = rowJson.find('{', keyPos);
101 if (objStart == std::string_view::npos) return "";
102
103 // Inside this object, find "value" key
104 size_t valueLabelPos = rowJson.find("\"value\"", objStart);
105 if (valueLabelPos == std::string_view::npos) return "";
106
107 // Find the colon after "value"
108 size_t colonPos = rowJson.find(':', valueLabelPos);
109 if (colonPos == std::string_view::npos) return "";
110
111 // Find the opening quote of the actual value, skipping whitespace/newlines
112 size_t valueStartQuote = std::string_view::npos;
113 for (size_t i = colonPos + 1; i < rowJson.length(); ++i) {
114 if (rowJson[i] == '"') {
115 valueStartQuote = i;
116 break;
117 }
118 }
119 if (valueStartQuote == std::string_view::npos) return "";
120
121 // Find the closing quote (handling escaped quotes is skipped for simplicity)
122 size_t valueEndQuote = rowJson.find('"', valueStartQuote + 1);
123 if (valueEndQuote == std::string_view::npos) return "";
124
125 return std::string(rowJson.substr(valueStartQuote + 1, valueEndQuote - valueStartQuote - 1));
126 }

211 template <typename T>
212 static std::vector<T> parseJsonResponse(const std::string& rawJson) {
213 std::vector<T> results;
214
215 // 1. Extract raw JSON strings for each row
216 auto rows = MiniSparqlParser::extractBindings(rawJson);
217
218 // 2. Iterate over rows and map to C++ struct using Reflection
219 for (const auto& rowJson : rows) {
220 T item{}; // Value initialization (important for ints to be 0)
221 using ReflectedType = T;
222 constexpr auto type_meta = ^^ReflectedType;
223
224 // Reflection Loop: Iterate over struct members
225 [:expand(meta::members_of(type_meta, meta::access_context::unchecked())):] >> [&]<auto member>{
226 if constexpr (meta::is_nonstatic_data_member(member)) {
227 // Get compile-time name of the member (e.g., "itemLabel")
228 constexpr auto nameView = meta::identifier_of(member);
229
230 // Extract value from JSON using our mini-parser
231 std::string value = MiniSparqlParser::extractValue(rowJson, nameView);
232
233 if (!value.empty()) {
234 // Get the type of the member to perform correct conversion
235 using MemberType = typename [: meta::type_of(member) :];
236
237 // Assign based on type
238 if constexpr (std::is_same_v<MemberType, std::string>) {
239 item.[:member:] = value;
240 }
241 else if constexpr (std::is_integral_v<MemberType>) {
242 // Convert string to integer type (int, long, size_t, etc.)
243 try {
244 item.[:member:] = static_cast<MemberType>(std::stoll(value));
245 } catch (...) {
246 // Ignore conversion errors in this simple parser
247 }
248 }
249 else if constexpr (std::is_floating_point_v<MemberType>) {
250 // Convert string to floating point (float, double)
251 try {
252 item.[:member:] = static_cast<MemberType>(std::stod(value));
253 } catch (...) {
254 // Ignore errors
255 }
256 }
257 }
258 }
259 };
260
261 results.push_back(item);
262 }
263 return results;
264 }
